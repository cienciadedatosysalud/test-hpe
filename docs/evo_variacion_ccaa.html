viewof indicador = Inputs.select(
  [ 'total_HPE', 'Dehydration_HPE', 'CHF_HPE','diabetes_HPE', 'Asthma_HPE','EPOC_HPE','Angina_HPE'],
  { label: 'Indicador',
  format: x => {
      switch (x) {
        case 'total_HPE':
          return 'Cualquier causa';
        case 'Dehydration_HPE':
          return 'Deshidratación';
        case 'CHF_HPE':
          return 'Insuficiencia Cardiaca Congestiva (ICC)';
        case 'diabetes_HPE':
          return 'Complicaciones Agudas de la Diabetes';
        case 'Asthma_HPE':
          return 'Asma en adulto';
        case 'EPOC_HPE':
          return 'Enfermedad Pulmonar Obstructiva Crónica (EPOC)';
        case 'Angina_HPE':
          return 'Angina (no primaria) sin procedimiento';
        default:
          return x; // Para valores no definidos
      }
    }}
);
